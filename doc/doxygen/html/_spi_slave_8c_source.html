92  GpioValue_DONT_CARE, GpioDriveStrength_DEFAULT);
93 
94  /* put the module in reset to configure it */
95  UCB1CTL1 = UCSWRST;
96 
97  /* data is captured on the first clock edge and changed on the following edge */
98  /* inactive clock state is low, MSB first*/
99  /* 8-bit transfer, Slave mode 4-pin SPI, chip select is active low */
100  UCB1CTL0 = UCCKPH | UCMSB | UCMODE_2 | UCSYNC;
101 
102  /* enable the SPI module */
103  UCB1CTL1 &= ~UCSWRST;

106  Dma_Configure(_dmaChannelTx, &_dmaConfigTx);

109  Dma_RegisterIsr(_dmaChannelTx, &_HandleDmaTxIsr);

129 SpiSlaveError_t SpiSlave_LoadPacket(uint16_t packetLength,
130  const uint8_t packet[])
131 {
133  static volatile uint8_t* const spiTxBuffer = &UCB1TXBUF;
134 
135  SpiSlaveError_t error = SpiSlaveError_OK;
136 
137  /* make sure the packet has > 1 element because the DMA needs to load a packet
138  to start the DMA trigger and have at least one element for the DMA to transfer.
139  */
140  ASSERT(packetLength > 1u, ErrorLedMode_SOFTWARE);
141  ASSERT(packet != NULL, ErrorLedMode_SOFTWARE);
142 
143  if (_bufferTxInUse == false)
144  {
145  _bufferTxInUse = true;
146  _bufferTxComplete = false;
147  Dma_Arm(_dmaChannelTx, (void*)&packet[1], (void*)spiTxBuffer, packetLength - 1);
148  Dma_EnableIsr(_dmaChannelTx);
149  Dma_Enable(_dmaChannelTx);
150  *spiTxBuffer = packet[0];
151  }
152  else
153  {
154  error = SpiSlaveError_BUFFER_IN_USE;
155  }
156 
157  return error;
158 }
