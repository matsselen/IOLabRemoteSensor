119 static bool _ProcessEvents_RUN(Event_t event)
120 {
121  RfStatus_t rfStatus;
122  bool continueProcessing = false;
123 
124  switch(event)
125  {
126  /* this event must be immediately before default because it falls through */
127  case Event_BUTTON_1_RELEASED:
128  CommProtocol_GetRfStatus(&rfStatus);
129  if ((_buttonOnePressed && PushButtonControl_isButtonPressed(PushButton_ZERO)) &&
130  ((rfStatus.connectionStatus == RfConnectionStatus_NOT_PAIRED) ||
131  (rfStatus.connectionStatus == RfConnectionStatus_PAIRED_NOT_CONNECTED)))
132  {
133  /* Before we enter the self-test, unlatch the system power so that the
134  self-test will automatically end if the power button is released */
135  _UnlatchPower();
136  _currentState = State_SELF_TEST;
137  EventQueue_Add(Event_START_SELF_TEST, EventPriority_LOWEST);
138  }
139  /* do not break. continue with any other processing of the event */
140 
141  default:

143  break;
144  }
