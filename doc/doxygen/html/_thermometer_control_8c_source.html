363 static void _thermometerDataCallBack(uint16_t data)
364 {
365  static uint16_t dataCopy[2u];
366 
367  _data += data;
368  if (++_numberAccumulated >= _numberToAccumulate)
369  {
370  dataCopy[0u] = (uint16_t)(_data>>16u);
371  dataCopy[1u] = (uint16_t)(_data);
372  DataBuffer_AddToSubarray16Bit(SensorId_THERMOMETER, 2u, dataCopy);
373 
374 #ifdef THERMOMETER_PRINT_TO_SERIAL /* use serial print for data */
375  Serial_SendStringNumberU32("Otemp:", _data);
376 #endif
377 
378  /* reset counters and data */
379  _numberAccumulated = 0u;
380  _data = 0u;
381  }
