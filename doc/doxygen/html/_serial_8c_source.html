179 /***************************************************************************//****************************************************************************/
182 void Serial_ProcessReceivedBytes(void)
183 {
184  static uint16_t currentPayloadIndex = 0u;
185 // static uint16_t payloadLength = 0u;
186  bool payloadInUse;
187  bool packetFound;
188  uint8_t byteRead;
189  UartError_t uartError;
190 
191  /* Only process bytes if the interface is open, receive mode is enabled, and
192  the payload structure is not in use.
193  */
194  payloadInUse = _serialPayloadRx.inUse;
195  if ((_isSerialInterfaceOpen) &&
196  (_currentMode == SerialMode_TRANSMIT_AND_RECEIVE) &&
197  (payloadInUse == false))
198  {
199  /* loop while bytes are available from the UART and a packet has not
200  been found.
201  */
202  packetFound = false;
203  while ((Uart_AreBytesAvailable(SERIAL_UART_PORT)) &&
204  (packetFound == false))
205  {
206  /* get next byte, if read error occurs, exit loop */
207  uartError = Uart_ReadByte(SERIAL_UART_PORT, &byteRead);
208  if (uartError != UartError_OK)
209  {
210  break;
211  }
212 
213  switch(_currentRxIndex)
214  {
215  case SERIAL_SOP_INDEX:
216  {
217  /* Look for start of packet byte. If SOP is binary, increment index
218  and initialize field storage value. If SOP is human, set index to
219  payload index since there is not more header for human readable.
220  */
221  if (byteRead == SERIAL_SOP_BINARY)
222  {
223  // TODO: spg - handle Binary SOP
224  }
225  else if (byteRead == SERIAL_SOP_HUMAN)
226  {
227  _currentRxIndex = SERIAL_PAYLOAD_INDEX;
228  currentPayloadIndex = 0u;
229  _serialPayloadRx.isBinary = false;
230  }
231  else
232  {
233  /* do nothing */
234  }
235  break;
236  }
237 
238  case SERIAL_PAYLOAD_INDEX:
239  {
240  if (_serialPayloadRx.isBinary)
241  {
242  // TODO: spg - handle binary packets
243  }
244  else
245  {
246  /* for human readable packets:
247  Save payload bytes until EOP byte is received or the payload
248  buffer becomes full. If an EOP is received, add a NULL character
249  to the end of the payload to make it a proper string. If necessary,
250  send an ACK. Return the pointer to the payload. Otherwise, if the
251  buffer is full and no EOP was found, send NAK and go back looking for
252  SOP byte.
253  */
254  if (byteRead == SERIAL_EOP)
255  {
256  _serialPayloadRx.inUse = true;
257  _serialPayloadRx.length = currentPayloadIndex;
258  _serialPayloadRx.buffer[currentPayloadIndex] = (uint8_t) '\0';
259 
260  packetFound = true;
261  EventQueue_Add(Event_SERIAL_RECEIVED_PACKET, EventPriority_LOWEST);
262 
263  /* restart the index looking for the start of the next packet */
264  _currentRxIndex = SERIAL_SOP_INDEX;
265 
266  #ifdef ENABLE_ACK_NAK_HUMAN_READABLE_PACKETS
267  _SendAck();
268  #endif
269  }
270  else
271  {
272  if (currentPayloadIndex < SERIAL_PAYLOAD_MAX_LENGTH_HUMAN)
273  {
274  _serialPayloadRx.buffer[currentPayloadIndex] = byteRead;
275  ++currentPayloadIndex;
276  }
277  else
278  {
279  _currentRxIndex = SERIAL_SOP_INDEX;
280 
281  #ifdef ENABLE_ACK_NAK_HUMAN_READABLE_PACKETS
282  _SendNak();
283  #endif
284  }
285  }
286 
287  }
288 
289  break;
290  }
291 
292  default:
293  /* should not reach default, flag error */
294  ASSERT(false, ErrorLedMode_SOFTWARE);
295  break;
296  }
297  }
298  }

345 SerialStatus_t Serial_SendPacket(const SerialBuffer_t* payloadToTransmit)
346 {
347  SerialStatus_t serialStatus = SerialStatus_SUCCESS;
348  UartError_t uartResult = UartError_OK;
349 
350  ASSERT(payloadToTransmit != NULL, ErrorLedMode_SOFTWARE);
351  ASSERT(_isSerialInterfaceOpen, ErrorLedMode_SOFTWARE);
352  ASSERT(payloadToTransmit->length <= SERIAL_PAYLOAD_MAX_LENGTH, ErrorLedMode_SOFTWARE);
353 
354  if (payloadToTransmit->isBinary)
355  {
356  // TODO: spg - handle binary packets
357  /* assert to notify that binary packets aren't working yet */
358  ASSERT(false, ErrorLedMode_SOFTWARE);
359  }
360  else
361  {
362  /* Send human readable packet in the form SOP - payload - footer. */
363  uartResult = Uart_WriteByte(SERIAL_UART_PORT, SERIAL_SOP_HUMAN);
364 
365  if (uartResult == UartError_OK)
366  {
367  uartResult = Uart_WriteByteArray(SERIAL_UART_PORT, payloadToTransmit->length,
368  payloadToTransmit->buffer);
369  }
370 
371  if (uartResult == UartError_OK)
372  {
373  uartResult = Uart_WriteByteArray(SERIAL_UART_PORT, FOOTER_HUMAN_LENGTH, _footerHuman);
374  }
375  }
376 
377  if (uartResult == UartError_BUFFER_FULL)
378  {
379  serialStatus = SerialStatus_TX_BUFFER_FULL;
380  }
381 
382  return serialStatus;
383 }
