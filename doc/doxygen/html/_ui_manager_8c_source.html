114 PlaySequenceStatus_t UiManager_PlaySequence(UiSequence_t sequence, PlayMode_t mode)
115 {
116  const UiSequenceData_t *newSequence;
117  PlaySequenceStatus_t status = PlaySequenceStatus_INVALID;
118  uint8_t desiredVolume = BUZZER_DUTY_MAX_VOLUME;
119 
120  switch(sequence)
121  {
122  case UiSequence_SUCCESS:
123  newSequence = &successSequence;
124  break;
125 
126  case UiSequence_FAIL:
127  newSequence = &failSequence;
128  break;
129 
130  case UiSequence_SPECIAL:
131  newSequence = &starWarsSequence;
132  break;
133 
134  case UiSequence_IN_PROGRESS:
135  desiredVolume = BUZZER_DUTY_MEDIUM_VOLUME;
136  newSequence = &girlILeftBehindSequence;
137  break;
138 
139  default:
140  status = PlaySequenceStatus_UNKNOWN_SEQUENCE;
141  break;
142  }
143 
144  if ((status != PlaySequenceStatus_UNKNOWN_SEQUENCE) &&
145  (_uiTimerActive == false))
146  {
147  Serial_SendStringNumber("UI: ", sequence);
148  _activeSequence.currentSequence = newSequence;
149  _playMode = mode;
150  _ZeroCurrentSequence(&_activeSequence);
151  _uiTimerActive = true;
152  _uiTimerHandle = TimerTick_RequestTimeout(TickTimeoutType_REPEAT, SEQUENCE_RESOLUTION_MS, &_UiTickCallBack);
153  Buzzer_SetDuty(desiredVolume);
154  _PlaySequence();
155  status = PlaySequenceStatus_SUCCESS;
156  }
157  else
158  {
159  DEBUG("UI in use");
160  status = PlaySequenceStatus_BUSY;
161  }
162

206 /***************************************************************************//****************************************************************************/
210 static void _PlaySequence(void)
211 {
212  volatile bool backlightStillRunning;
213  volatile bool buzzerStillRunning;
214  const BackLightSequenceStep_t *backlightStep;
215  const BuzzerSequenceStep_t *buzzerStep;
216  static bool buzzerEnabled = false;
217 
218  ASSERT(_activeSequence.currentSequence != NULL, ErrorLedMode_SOFTWARE);
219 
220  if (_activeSequence.endOfBacklightStep <= _activeSequence.elapsedMs)
221  {
222  /* make sure we aren't on the last step */
223  if (_activeSequence.backlightIndex < _activeSequence.currentSequence->backLightSequence.stepsInSequence)
224  {
225  backlightStep = &_activeSequence.currentSequence->backLightSequence.stepList[_activeSequence.backlightIndex];
226  _activeSequence.endOfBacklightStep = _activeSequence.elapsedMs + backlightStep->durationMs;
227  if (backlightStep->backlightOn)
228  {
229 #ifndef __DISABLE_BACKLIGHT
230  Backlight_Enable();
231 #endif
232  }
233  else
234  {
235 #ifndef __DISABLE_BACKLIGHT
236  Backlight_Disable();
237 #endif
238  }
239 
240  ++_activeSequence.backlightIndex;
241  backlightStillRunning = true;
242  }
243  else
244  {
245  /* we are on the last step and its timer has expired */
246  backlightStillRunning = false;
247 #ifndef __DISABLE_BACKLIGHT
248  Backlight_Enable();
249 #endif
250  }
251  }
252  else
253  {
254  backlightStillRunning = true;
255  }
256 
257  if (_activeSequence.endOfBuzzerStep <= _activeSequence.elapsedMs)
258  {
259  /* make sure we aren't on the last step */
260  if (_activeSequence.buzzerIndex < _activeSequence.currentSequence->buzzerSequence.stepsInSequence)
261  {
262  buzzerStep = &_activeSequence.currentSequence->buzzerSequence.stepList[_activeSequence.buzzerIndex];
263  _activeSequence.endOfBuzzerStep = _activeSequence.elapsedMs + buzzerStep->durationMs;
264  if (buzzerStep->enabled)
265  {
266  Buzzer_SetFrequency(buzzerStep->pitch);
267  /* don't re-enable the buzzer if it is already enabled */
268  if (buzzerEnabled == false)
269  {
270  buzzerEnabled = true;
271  Buzzer_Enable();
272  }
273  }
274  else
275  {
276  buzzerEnabled = false;
277  Buzzer_Disable();
278  }
279 
280  ++_activeSequence.buzzerIndex;
281  buzzerStillRunning = true;
282  }
283  else
284  { /* we are on the last step and its timer has expired */
285  buzzerStillRunning = false;
286  buzzerEnabled = false;
287  Buzzer_Disable();
288  }
289  }
290  else
291  {
292  buzzerStillRunning = true;
293  }
294 
295  /* If the backlight and the buzzer are done, then the sequence is complete */
296  if ((backlightStillRunning == false) &&
297  (buzzerStillRunning == false))
298  {
299  if (_playMode == PlayMode_SINGLE)
300  {
301  if (_callbackRunning)
302  {
303  TimerTick_CancelTimeoutFromCallback();
304  }
305  else
306  {
307  TimerTick_CancelTimeout(_uiTimerHandle);
308  }
309  _uiTimerActive = false;
310  _activeSequence.currentSequence = NULL;
311 
312  /* Enqueue and event to notify the system that the sequence is complete in case anything
313  was waiting for the sequence to complete. */
314  EventQueue_Add(EventApplication_SEQUENCE_COMPLETE, EventPriority_LOWEST);
315  }
316  else
317  {
318  /* Reset the sequence so it will play again */
319  _ZeroCurrentSequence(&_activeSequence);
320  }
321  }
322  else
323  {
324  _activeSequence.elapsedMs += SEQUENCE_RESOLUTION_MS;
325  }
326 }
