116 void TimerTick_ProcessTimeouts(void)
117 {
118  Timeout_t* timeout;
119  TickTimeoutType_t type;
120 
121  /* Only increment the counter and check for expired timeouts
122  if timeouts are pending
123  */
124  if (_startOfList != NULL)
125  {
126  ++_tickCounter;
127 
128  /* Find all timeouts that have expired */
129  timeout = _startOfList;
130  while ((timeout != NULL) && (_tickCounter == timeout->tickCount))
131  {
132  /* Call the timeout's callback */
133  if (timeout->callback != NULL)
134  {
135  _isCallbackRunning = true;
136 
137  (*(timeout->callback))();
138 
139  _isCallbackRunning = false;
140  }
141 
142  /* Delete timeout from the list.
143  Since the list is sorted, this timeout is always the first one
144  in the list.
145  */
146  _startOfList = timeout->next;
147  timeout->isActive = false;
148 
149  /* Re-insert repeat timeouts */
150  type = timeout->type;
151  if (type == TickTimeoutType_REPEAT)
152  {
153  _InsertTimeout(timeout);
154  }
155 
156  timeout = _startOfList;
157  }
158  }

176  uint16_t milliseconds,
177  TickTimeoutCallback_t callback)
178 {

181  uint16_t numTicks;

184

186  ASSERT(callback != NULL, ErrorLedMode_SOFTWARE);
187 
188  /* Disable timeout ISR to prevent modifications to the timeout list. */
189  TIMER_A1_CC0_INT_DISABLE();
190 
191  /* Get pointer to available storage */
192  newTimeout = NULL;
193  for (index = 0u; index < MAX_NUM_TIMEOUTS; ++index)
194  {
195  isTimeoutActive = _timeoutList[index].isActive;
196  if (isTimeoutActive == false)
197  {
198  newTimeout = &_timeoutList[index];
199  handle = (TickTimeoutHandle_t) index;
200 
201  /* Track the maximum number of timeout structures being used. */
202  if ((index + 1u) > _maxActiveTimeouts)
203  {
204  _maxActiveTimeouts = index + 1u;
205  }
206  break;
207  }
208  }
209 
210  if (newTimeout != NULL)
211  {
212  /* Convert milliseconds to ticks */
213  numTicks = (milliseconds + (TIMERTICK_MSEC_PER_TICK / 2u)) / TIMERTICK_MSEC_PER_TICK;
214 
215  /* Store timeout information */
216  newTimeout->type = type;
217  newTimeout->periodTicks = numTicks;
218  newTimeout->callback = callback;
219 
220  /* Insert timeout into sorted list */
221  _InsertTimeout(newTimeout);
222  }
223  else
224  {
225  /* No available storage for a new timeout */
226  ASSERT(false, ErrorLedMode_SOFTWARE);
227  }
228 
229  /* Always enable the timeout interrupt so that the timer is shown as
230  active, which causes the watchdog timer to be reset.
231  */
232  TIMER_A1_CC0_INT_ENABLE();
233

246 void TimerTick_CancelTimeout(TickTimeoutHandle_t handle)
247 {
248  Timeout_t* timeoutToDelete;
249  Timeout_t* timeout;
250  Timeout_t* previousTimeout;
251 
252  ASSERT(handle < MAX_NUM_TIMEOUTS, ErrorLedMode_SOFTWARE);
253  ASSERT(_isCallbackRunning == false, ErrorLedMode_SOFTWARE);
254 
255  timeoutToDelete = &_timeoutList[handle];
256 
257  /* Timeout only needs to be canceled if it is active */
258  if (timeoutToDelete->isActive)
259  {
260  /* Disable timeout ISR to prevent modifications to the timeout list. */
261  TIMER_A1_CC0_INT_DISABLE();
262 
263  /* Find the location of the timeout before the timeout to delete */
264  previousTimeout = NULL;
265  timeout = _startOfList;
266  while ((timeout != NULL) && (timeout != timeoutToDelete))
267  {
268  previousTimeout = timeout;
269  timeout = timeout->next;
270  }
271 
272  /* Assert that the timeout-to-delete was found in the list. */
273  ASSERT(timeout != NULL, ErrorLedMode_SOFTWARE);
274 
275  /* Delete timeout from the list.
276  timeout == timeoutToDelete at this point
277  */
278  if (previousTimeout == NULL)
279  {
280  /* Timeout is the first item in the list. */
281  _startOfList = timeout->next;
282  }
283  else
284  {
285  /* Timeout is NOT the first item in the list */
286  previousTimeout->next = timeout->next;
287  }
288  timeout->isActive = false;
289 
290  /* Always enable the timeout interrupt so that the timer is shown as
291  active, which causes the watchdog timer to be reset.
292  */
293  TIMER_A1_CC0_INT_ENABLE();
294  }

415 static void _InsertTimeout(Timeout_t* const newTimeout)
416 {
417  uint16_t periodTicks;
418  uint16_t tickCount;
419  uint16_t currentTickCount;
420  Timeout_t* timeout;
421  Timeout_t* nextLatestTimeout;
422  Timeout_t* previousTimeout;
423 
424  /* Determine next tick count when the timeout will occur.
425  Adjust the tick counter if the timeout period causes the tick counter
426  to rollover.
427  */
428  periodTicks = newTimeout->periodTicks;
429  currentTickCount = _tickCounter;
430  if (periodTicks > (0xFFFFu - currentTickCount))
431  {
432  /* Reduce the current tickCount of every active timeout by the
433  current tick count and reset the _tickCounter to zero.
434  Each tickCount will be greater than the current tick count,
435  otherwise the timeout would have expired.
436  */
437  timeout = _startOfList;
438  while (timeout != NULL)
439  {
440  tickCount = timeout->tickCount;
441  timeout->tickCount = tickCount - currentTickCount;
442 
443  timeout = timeout->next;
444  }
445  currentTickCount = 0u;
446  _tickCounter = 0u;
447  }
448 
449  newTimeout->tickCount = periodTicks + currentTickCount;
450 
451 /* The compiler incorrectly flags MISRA rule 17.6 in the following code.
452  The rule states that the address of an automatic variable should not be
453  stored. The compiler flags the storage of the newTimeout pointer in
454  static variables. This is not a violation, as only the pointer is an
455  automatic variable, not the object it points to.
456 */
457 #pragma CHECK_MISRA("-17.6")
458  /* If the timeout list is empty, start new list.
459  Otherwise, search list for correct location.
460  */
461  if (_startOfList == NULL)
462  {
463  newTimeout->next = NULL;
464  _startOfList = newTimeout;
465  }
466  else
467  {
468  /* Find first timeout that expires after new timeout */
469  nextLatestTimeout = _startOfList;
470  previousTimeout = NULL;
471  while ((nextLatestTimeout->tickCount <= newTimeout->tickCount) &&
472  (nextLatestTimeout->next != NULL))
473  {
474  previousTimeout = nextLatestTimeout;
475  nextLatestTimeout = nextLatestTimeout->next;
476  }
477 
478  if (nextLatestTimeout->tickCount <= newTimeout->tickCount)
479  {
480  /* Reached last item in the list.
481  Insert new timeout after the last one.
482  */
483  nextLatestTimeout->next = newTimeout;
484  newTimeout->next = NULL;
485  }
486  else
487  {
488  /* Found next latest timeout in the list. Insert the new
489  timeout before this one.
490  */
491  if (previousTimeout == NULL)
492  {
493  _startOfList = newTimeout;
494  }
495  else
496  {
497  previousTimeout->next = newTimeout;
498  }
499  newTimeout->next = nextLatestTimeout;
500  }
501  }
502 #pragma RESET_MISRA("all")
503 
504  /* Indicate that timeout is active */
505  newTimeout->isActive = true;
506 }
