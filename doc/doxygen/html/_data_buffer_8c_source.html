352  uint16_t endIndex;
353  uint16_t dataIndex;
354  uint16_t writeIndex;

356  /* find the index of the subarray id */
357  subarrayIndex = _IndexOfSubarrayId(subArrayId);
358 
359  if (subarrayIndex != INVALID_INDEX)
360  {
361  startIndex = _subArrayStartIndex[subarrayIndex];
362  currentLength = _writeArray[startIndex + SUBARRAY_LENGTH_BYTE_OFFSET];
363  endIndex = _subArrayEndIndex[subarrayIndex];
364  /* Length must be doubled because data is Uint16_t and buffer is Uint8_t */
365  if ((startIndex + currentLength + (length * 2)) < endIndex)
366  {
367  writeIndex = startIndex + SUBARRAY_DATA_BYTE_OFFSET + currentLength;
368  for (dataIndex = 0; dataIndex < length; ++dataIndex)
369  {
370  /* Take the MSB first, then the LSB */
371  _writeArray[writeIndex] = ((uint8_t)(data[dataIndex] >> 8u));
372  ++writeIndex;
373  _writeArray[writeIndex] = ((uint8_t)(data[dataIndex]));
374  ++writeIndex;
375  }
376  /* length is uint16_t so needs to be doubled */
377  _writeArray[startIndex + SUBARRAY_LENGTH_BYTE_OFFSET] += length * 2;
378 
379  addSuccess = true;
380  }
381  else
382  {
383  /* set the overflow bit */
384  _writeArray[startIndex + SUBARRAY_ID_BYTE_OFFSET] |= OVERFLOW_BIT;
385  }
