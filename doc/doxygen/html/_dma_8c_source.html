103 void Dma_Init( void )
104 {
105  /* Disable the DMA from interrupting read-modify-write operations by the CPU.
106  This is a workaround for DMA4 Errata, in which a write to a 20-bit address
107  is corrupted by any DMA transfer.
108  */
109  DMACTL4 = DMARMWDIS;
110  _dmaInitialized = true;
111 }

113 /***************************************************************************//****************************************************************************/
120 void Dma_Configure(DmaChannel_t channel,
121  const DmaConfig_t *config)
122 {
124  #define DMA_CHANNELS_PER_TRIG_REG 2
125 
126  ASSERT(_dmaInitialized, ErrorLedMode_SOFTWARE);
127  ASSERT(channel < DMA_NUM_CHANNELS, ErrorLedMode_SOFTWARE);
128  ASSERT(config != NULL, ErrorLedMode_SOFTWARE);
129 
130  volatile uint16_t* dmaCtrlReg = _dmaControlRegisters[channel];
131 
132  /* each trigger register is for 2 channels, so the trigger register is
133  the DMA channel / 2. If the channel number is odd the configuration is
134  in the high byte.
135  */
136  volatile uint16_t* dmaTrigReg = _dmaTriggerRegisters[channel/DMA_CHANNELS_PER_TRIG_REG];
137 
138  /* Ensure that the DMA is disabled before changing its configuration */
139  *dmaCtrlReg = 0;
140 
141  /* set the DMA mode */
142  *dmaCtrlReg |= config->mode;
143 
144  /* set the DMA source and destination increment and size */
145  *dmaCtrlReg |= (((uint16_t)(config->sourceIncr)) * DMASRCINCR0) +
146  (((uint16_t)(config->sourceBitWidth)) * DMASRCBYTE) +
147  (((uint16_t)(config->destinationIncr)) * DMADSTINCR0) +
148  (((uint16_t)(config->destinationBitWidth)) * DMADSTBYTE);
149 
150  /* set the trigger source */
151  if ((channel % 2) == 0) /* is even */
152  {
153  /* even channels are in the low byte of the register. Make sure old value
154  is clear before setting new value
155  */
156  *dmaTrigReg = ((uint16_t)(*dmaTrigReg & 0xFF00)) | ((uint16_t)(config->trigger));
157  }
158  else /* is odd */
159  {
160  /* odd channels are in the high byte of the register. Make sure old value
161  is clear before setting new value
162  */
163  *dmaTrigReg = ((uint16_t)(*dmaTrigReg & 0x00FF)) | (((uint16_t)(config->trigger))<<8u);
164  }
165 
166  _dmaConfigured[channel] = true;
167 }

319 __interrupt void _DmaIsr(void)
320 {
321  switch (_even_in_range(DMAIV, DMAIV_DMA2IFG))
322  {
323  case DMAIV_DMA0IFG:
324  if (_dmaIsrList[0] != NULL)
325  {
326  (*_dmaIsrList[0])();
327  }
328  break;
329 
330  case DMAIV_DMA1IFG:
331  if (_dmaIsrList[1] != NULL)
332  {
333  (*_dmaIsrList[1])();
334  }
335  break;
336 
337  case DMAIV_DMA2IFG:
338  if (_dmaIsrList[2] != NULL)
339  {
340  (*_dmaIsrList[2])();
341  }
342  break;
343 
344  case DMAIV_NONE:
345  /* If the interrupt is being disabled as the interrupt is firing, the ISR
346  could get called when no flags are set. Need to handle that here.
347  */
348  break;
349 
350  default:
351  _never_executed();
352  break;
353  }
