300 void AdcManager_EnableChannel(
301  AdcManChannel_t channel,
302  AdcManSampleRate_t rate,
303  AdcManRef_t reference,
304  AdcDataCallback_t callback)
305 {
306  bool firstEnabledChannel;
307  AdcChannelData_t *storedChannel;
308  uint16_t channelCounter;
309  AdcSamplesNeeded_t samplesNeeded;
310 
311  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
312  ASSERT((channel >= AdcManChannel_FIRST) && (channel < AdcManChannel_LAST), ErrorLedMode_SOFTWARE);
313  ASSERT((rate >= AdcManSampleRate_FIRST) && (rate < AdcManSampleRate_LAST), ErrorLedMode_SOFTWARE);
314  ASSERT((callback != NULL), ErrorLedMode_SOFTWARE);
315 
316  firstEnabledChannel = _AllChannelsDisabled();
317 
318  /* High speed sampling ( > 400 Hz) */
319  if ((rate > AdcManSampleRate_0400HZ) && (rate < AdcManSampleRate_LAST))
320  {
321  switch (rate)
322  {
323  case AdcManSampleRate_0800HZ:
324  /* fast mode can only handle a single sample rate, make sure they are the same */
325  if (_sampleMode == AdcManSampleMode_FAST)
326  {
327  ASSERT(_timerRate == AdcTimerRate_0800HZ, ErrorLedMode_SOFTWARE);
328  }
329  _timerRate = AdcTimerRate_0800HZ;
330  samplesNeeded = AdcSamplesNeeded_0800HZ;
331  break;
332 
333  case AdcManSampleRate_2400HZ:
334  /* fast mode can only handle a single sample rate, make sure they are the same */
335  if (_sampleMode == AdcManSampleMode_FAST)
336  {
337  ASSERT(_timerRate == AdcTimerRate_2400HZ, ErrorLedMode_SOFTWARE);
338  }
339  _timerRate = AdcTimerRate_2400HZ;
340  samplesNeeded = AdcSamplesNeeded_2400HZ;
341  break;
342 
343  case AdcManSampleRate_4800HZ:
344  /* fast mode can only handle a single sample rate, make sure they are the same */
345  if (_sampleMode == AdcManSampleMode_FAST)
346  {
347  ASSERT(_timerRate == AdcTimerRate_4800HZ, ErrorLedMode_SOFTWARE);
348  }
349  _timerRate = AdcTimerRate_4800HZ;
350  samplesNeeded = AdcSamplesNeeded_4800HZ;
351  break;
352 
353  case AdcManSampleRate_6000HZ:
354  /* fast mode can only handle a single sample rate, make sure they are the same */
355  if (_sampleMode == AdcManSampleMode_FAST)
356  {
357  ASSERT(_timerRate == AdcTimerRate_6000HZ, ErrorLedMode_SOFTWARE);
358  }
359  _timerRate = AdcTimerRate_6000HZ;
360  samplesNeeded = AdcSamplesNeeded_6000HZ;
361  break;
362 
363  default:
364  DEBUG("invalid rate");
365  break;
366  }
367  /* must come after previous switch, to handle the transition into fast mode */
368  if (_sampleMode != AdcManSampleMode_FAST)
369  {
370  _sampleMode = AdcManSampleMode_FAST;
371  _highSpeedActiveChannels = 1;
372  Adc_Config(&_fastAdcConfig);
373  }
374  else /* alread in high speed mode */
375  {
376  ++_highSpeedActiveChannels;
377  /* in order to add a channel in high speed mode, the ADC first needs to be stopped. Use the
378  first enabled variable to turn the ADC back on */
379  _AdcStop();
380  firstEnabledChannel = true;
381  /* if multiple channels are configured for high speed, use the fast multi mode */
382  Adc_Config(&_fastMultiAdcConfig);
383  samplesNeeded = (AdcSamplesNeeded_t)1u;
384  }
385 
386  /* set all the channels to be used for single channel high speed */
387  for (channelCounter = ((_highSpeedActiveChannels - 1u)*(uint16_t)samplesNeeded); channelCounter < ((_highSpeedActiveChannels)*(uint16_t)samplesNeeded); ++channelCounter)
388  {
389  _adcAllChannelConfig[channelCounter].inputChannel = (AdcInputChannel_t)(channel);
390  if ((AdcInputChannel_t)(channel) == AdcInputChannel_TEMP)
391  {
392  _adcAllChannelConfig[channelCounter].reference = ADC_THERMOMETER_REFERENCE;
393  }
394  else
395  {
396  _adcAllChannelConfig[channelCounter].reference = AdcManager_AdcManRefToAdcRef(reference);
397  }
398  storedChannel = &_channels[channelCounter];
399 
400  storedChannel->callback = callback;
401  storedChannel->skipSamples = _GetSamplesToSkip(rate);
402  storedChannel->skippedSamples = 0u;
403  }
404 
405  /* make sure all channels after this last added channel are not in use */
406  for (channelCounter = channelCounter; channelCounter < (uint16_t)AdcManChannel_LAST; ++channelCounter)
407  {
408  _channels[channelCounter].callback = NULL;
409  }
410 
411  _SetupTimerConfig();
412  }
413  /* Normal speed ADC sampling ( <= 400 Hz) */
414  else if ((rate >= AdcManSampleRate_FIRST) && (rate <= AdcManSampleRate_0400HZ))
415  {
416  if (_sampleMode != AdcManSampleMode_NORMAL)
417  {
418  _SetupAdcForNormalMode(reference);
419  }
420  else
421  {
422  _SetupAdcSingleChannel(channel, reference);
423  }
424 
425  storedChannel = &_channels[channel];
426 
427  storedChannel->callback = callback;
428  storedChannel->skipSamples = _GetSamplesToSkip(rate);
429  storedChannel->skippedSamples = 0u;
430  }
431  else
432  {
433  DEBUG("invalid sample rate");
434  }
435 
436  AdcManager_PoweredChannelControl(channel, true);
437 
438  if (firstEnabledChannel)
439  {
440  _AdcStart();
441  }
442  else
443  {
444  DEBUG("adc running");
445  }
446 }

455 void AdcManager_DisableChannel(AdcManChannel_t channel)
456 {
457  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
458  ASSERT((channel >= AdcManChannel_FIRST) && (channel < AdcManChannel_LAST), ErrorLedMode_SOFTWARE);
459 
460  uint16_t channelCounter;
461  bool channelWasDisabled = false;
462 
463  if (_sampleMode != AdcManSampleMode_NORMAL)
464  {
465  /* all channels could be set to a single input for high speed sampling */
466  for (channelCounter = 0; channelCounter < AdcManChannel_LAST; ++channelCounter)
467  {
468  /* disable channels whose input is the same as the disabled channel */
469  if (_adcAllChannelConfig[channelCounter].inputChannel == (AdcInputChannel_t)channel)
470  {
471  _channels[channelCounter].callback = NULL;
472  channelWasDisabled = true;
473  }
474  }
475 
476  if (channelWasDisabled)
477  {
478  --_highSpeedActiveChannels;
479  }
480  }
481  else
482  {
483  _channels[channel].callback = NULL;
484  }
485 
486  AdcManager_PoweredChannelControl(channel, false);
487 
488  if (_AllChannelsDisabled())
489  {
490  _AdcStop();
491 
492  /* always return to the NORMAL ADC mode when finished sampling. */
493  _SetupAdcForNormalMode(AdcManRef_DEFAULT);
494  }
495  else
496  {
497  DEBUG("adc stopped");
498  }

542  _SetupTimerConfig();
543  }

706 /***************************************************************************//****************************************************************************/
710 void _SetupTimerConfig(void)
711 {
712  uint16_t numberChannelsNeeded;
713 
714  /* stop the timer before configuring it */
715  TIMER_A0_STOP();
716 
717  /* Configure the timer to run at 4Hz. ACLK/64 = 187.5kHz, 46875 count = 250ms */
718  /* Configure the timer to run at 100Hz. ACLK/2 = 6MHz, 60000 count = 10ms */
719  /* Configure the timer to run at 400Hz. ACLK/2 = 6MHz, 15000 count = 2.5ms */
720  /* Configure the timer to run at 800Hz. ACLK/2 = 6MHz, 7500 count = 1.25ms */
721  /* Configure the timer to run at 2400Hz. ACLK/2 = 6MHz, 2500 count = 416.6us */
722  /* Configure the timer to run at 4800Hz. ACLK/2 = 6MHz, 1250 count = 208.3us */
723  /* Configure the timer to run at 6kHz. ACLK/2 = 6MHz, 1000 count = 166.7us */
724 
725  /* NOTE: Set the number of channels such that the interrupt to service all the conversions
726  doesn't occur faster than 400 Hz.
727  */
728 
729  switch (_timerRate)
730  {
731  case AdcTimerRate_0400HZ:
732  Timer_A0_Configure(TimerSource_ACLK, TimerDivider_2, TimerRunMode_UP, 15000u - 1u);
733  Timer_A0CCX_Configure(TimerA0CC_1, OUTMOD_3, 1000u - 1u);
734  numberChannelsNeeded = (_highSpeedActiveChannels > 1) ? _highSpeedActiveChannels : AdcSamplesNeeded_0400HZ * _highSpeedActiveChannels;
735  _InitialzeXAdcChannels(numberChannelsNeeded);
736  break;
737 
738  case AdcTimerRate_0800HZ:
739  Timer_A0_Configure(TimerSource_ACLK, TimerDivider_2, TimerRunMode_UP, 7500 - 1u);
740  Timer_A0CCX_Configure(TimerA0CC_1, OUTMOD_3, 1000u - 1u);
741  numberChannelsNeeded = (_highSpeedActiveChannels > 1) ? _highSpeedActiveChannels : AdcSamplesNeeded_0800HZ * _highSpeedActiveChannels;
742  /* 800Hz / 100Hz = 2 samples per frame, 2 sample fifo = 400Hz interrupt rate */
743  _InitialzeXAdcChannels(numberChannelsNeeded);
744  break;
745 
746  case AdcTimerRate_2400HZ:
747  Timer_A0_Configure(TimerSource_ACLK, TimerDivider_2, TimerRunMode_UP, 2500u - 1u);
748  Timer_A0CCX_Configure(TimerA0CC_1, OUTMOD_3, 1000u - 1u);
749  numberChannelsNeeded = (_highSpeedActiveChannels > 1) ? _highSpeedActiveChannels : AdcSamplesNeeded_2400HZ * _highSpeedActiveChannels;
750  /* 2400Hz / 100Hz = 24 samples per frame, 6 sample fifo = 400Hz interrupt rate */
751  _InitialzeXAdcChannels(numberChannelsNeeded);
752  break;
753 
754  case AdcTimerRate_4800HZ:
755  Timer_A0_Configure(TimerSource_ACLK, TimerDivider_2, TimerRunMode_UP, 1250u - 1u);
756  Timer_A0CCX_Configure(TimerA0CC_1, OUTMOD_3, 1000u - 1u);
757  numberChannelsNeeded = (_highSpeedActiveChannels > 1) ? _highSpeedActiveChannels : AdcSamplesNeeded_4800HZ * _highSpeedActiveChannels;
758  /* 4800Hz / 100Hz = 48 samples per frame, 12 sample fifo = 400Hz interrupt rate */
759  _InitialzeXAdcChannels(numberChannelsNeeded);
760  break;
761 
762  case AdcTimerRate_6000HZ:
763  Timer_A0_Configure(TimerSource_ACLK, TimerDivider_2, TimerRunMode_UP, 1000u - 1u);
764  Timer_A0CCX_Configure(TimerA0CC_1, OUTMOD_3, 1000u - 1u);
765  numberChannelsNeeded = (_highSpeedActiveChannels > 1) ? _highSpeedActiveChannels : AdcSamplesNeeded_6000HZ * _highSpeedActiveChannels;
766  /* 6000Hz / 100Hz = 60 samples per frame, 15 sample fifo = 400Hz interrupt rate */
767  _InitialzeXAdcChannels(numberChannelsNeeded);
768  break;
769 
770  default:
771  ASSERT(false, ErrorLedMode_SOFTWARE);
772  break;
773 
774  }

789 {
790  switch(channel)

793  _poweredSensorsEnabled = enable ?
794  (AdcPoweredChannel_t)(_poweredSensorsEnabled | AdcPoweredChannel_FORCE_GAUGE) :
