70 void UltrasoundStartTransmit(void)
71 {
72  uint8_t pulseNum;
73 
74  _TxEnable();
75 
76  /* Disable interrupts so they don't affect the clock cycle counting */
77  INTERRUPT_DISABLE();
78  /* Create burst of 8 40kHz pulses every 10ms */
79  while (1)
80  {
81  /* pet the watchdog so it doesn't reset the system in our loop */
82  WATCHDOG_RESET();
83 
84  /* Change direction pin during delay */
85  GPIO_CLEAR(ULTRASONIC_DIRECTION_PORT, ULTRASONIC_DIRECTION_PIN);
86 
87  CLOCK_DELAY_US(ULTRASOUND_DELAY_US);
88 
89  GPIO_SET(ULTRASONIC_DIRECTION_PORT, ULTRASONIC_DIRECTION_PIN);
90 
91  for(pulseNum = 0; pulseNum < ULTRASOUND_NUM_PULSES; pulseNum++)
92  {
93  GPIO_SET(ULTRASONIC_OUTPUT_PORT, ULTRASONIC_OUTPUT_PIN);
94  CLOCK_DELAY_US(ULTRASOUND_HALFPERIOD_US);
95  GPIO_CLEAR(ULTRASONIC_OUTPUT_PORT, ULTRASONIC_OUTPUT_PIN);
96  CLOCK_DELAY_US(ULTRASOUND_HALFPERIOD_US);
97  }
98  }
99 }
