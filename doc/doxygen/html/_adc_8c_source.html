188 /***************************************************************************//****************************************************************************/
195 void Adc_Config( const AdcConfig_t *config)
196 {
197  /* Disable ADC conversion interrupts */
198  ADC12IE = 0u;
199 
200  /* ADC12 control 0:
201  Don't care about reference voltage
202  Reference generator off
203  ADC on
204  Stop conversions and disable overflow interrupts
205  */
206  ADC12CTL0 = ADC12ON;
207 
208  /* mask off and set the sample and hold time */
209  ADC12CTL0 |= (config->sampleHoldTime * ADC12SHT10) & (ADC12SHT13|ADC12SHT12|ADC12SHT11|ADC12SHT10);
210  ADC12CTL0 |= (config->sampleHoldTime * ADC12SHT00) & (ADC12SHT03|ADC12SHT02|ADC12SHT01|ADC12SHT00);
211 
212  /* use the config to set the multiple sample and conversion enable */
213  if (config->multipleSampleConversion)
214  {
215  ADC12CTL0 |= ADC12MSC;
216  }
217  else
218  {
219  ADC12CTL0 &= ~ADC12MSC;
220  }
221 
222  /* ADC12 control 1:
223  Start with first conversion in sequence
224  Use TA0.1 compare bit to start conversions
225  Sample and hold signal from the sample timer
226  Do not invert sample-and-hold signal
227  Divide clock source by 1
228  Use ACLK as source (12 MHz)
229  */
230  ADC12CTL1 = ADC12CSTARTADD_0 | ADC12SHS_1 | ADC12SHP |
231  ADC12DIV_0 | ADC12SSEL_1;
232 
233  /* clear the conversion sequence mode select */
234  ADC12CTL1 &= ~(ADC12CONSEQ1 | ADC12CONSEQ0);
235  /* mask off the config use it to set the conversion sequence mode select */
236  ADC12CTL1 |= (config->conSeq * ADC12CONSEQ0) & (ADC12CONSEQ1 | ADC12CONSEQ0);
237 
238  /* ADC12 control 2:
239  Pre-divide the clock source by 4 (12MHz / 4 = 3MHz)
240  Temperature sensor controlled by REFCTL0
241  12-bit ADC resolution
242  Binary data format
243  Reference buffer supports slow sampling rate
244  Reference output controlled by REFCTL0
245  Reference buffer in burst mode
246  */
247  ADC12CTL2 = ADC12PDIV | ADC12RES_2 | ADC12SR | ADC12REFBURST;
248 }

546 /***************************************************************************//****************************************************************************/
557 int16_t Adc_ConvertValueToTemp(uint16_t value)
558 {
559  int16_t calValueAt30C = 0;
560  int16_t tempGainFactor = 0;
561  int16_t valueSigned;
562  int16_t temperature;
563 
564  ASSERT(value <= ADC_MAX_RESULT, ErrorLedMode_SOFTWARE);
565  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
566 
567  /* Get the required values for the selected reference level */
568  switch (_selectedReferenceLevel)
569  {
570  case AdcReferenceLevel_1p5_VOLTS:
571  calValueAt30C = (int16_t) _adcCalibrationInfo.temp30Ref15V;
572  tempGainFactor = (int16_t) _temperatureGainFactorRef15V;
573  break;
574 
575  case AdcReferenceLevel_2p0_VOLTS:
576  calValueAt30C = (int16_t) _adcCalibrationInfo.temp30Ref20V;
577  tempGainFactor = (int16_t) _temperatureGainFactorRef20V;
578  break;
579 
580  case AdcReferenceLevel_2p5_VOLTS:
581  calValueAt30C = (int16_t) _adcCalibrationInfo.temp30Ref25V;
582  tempGainFactor = (int16_t) _temperatureGainFactorRef25V;
583  break;
584 
585  case AdcReferenceLevel_DONT_CARE:
586  ASSERT(false, ErrorLedMode_SOFTWARE);
587  break;
588 
589  default:
590  ASSERT(false, ErrorLedMode_SOFTWARE);
591  break;
592  }
593 
594  /* The equation used to relate the ADC value (sensor voltage) to temperature
595  (in Celsius) is as follows:
596  Vsense = TCsensor x Temp + Vsensor@0C
597  where TCsensor is the sensor's gain (mV/degC)
598  Vsensor@0C is the sensor voltage at 0C
599  Vsense is the sensor voltage at Temp
600 
601  In order to use the calibration values taken at 30C and 85C, the equation
602  can be modified as follows:
603  Cal@85 - Cal@30
604  Vsense = ----------------- x (Temp - 30) + Cal@30
605  85 - 30
606 
607  Solving for the temperature gives:
608  85 - 30
609  Temp = ----------------- x (Vsense - Cal@30) + 30
610  Cal@85 - Cal@30
611 
612  A scale factor is applied to the gain to make the integer math more
613  accurate and then removed at the end of the calculation. Before dividing
614  by the scale factor, half of the scale factor is added to the dividend
615  (or subtracted from the dividend if it is negative) to round to the nearest.
616  */
617 
618  valueSigned = (int16_t) value;
619  temperature = valueSigned - calValueAt30C;
620  temperature = temperature * tempGainFactor;
621 
622  if (temperature >= 0)
623  {
624  temperature = (temperature + (ADC_TEMP_SCALE_FACTOR / 2)) / ADC_TEMP_SCALE_FACTOR;
625  }
626  else
627  {
628  temperature = (temperature - (ADC_TEMP_SCALE_FACTOR / 2)) / ADC_TEMP_SCALE_FACTOR;
629  }
630 
631  temperature = temperature + 30;
632 
633  return temperature;
634 }

645 void Adc_ApplyCorrection(
646  AdcCorrection_t correction,
647  uint16_t numValuesToCorrect,
648  const uint16_t rawValues[],

651  uint16_t refCalFactor = 0u;
652  uint16_t value;
653  uint16_t i;
654  int16_t resultSigned;
655 
656  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
657 
658  /* Get the adjustment factor for the selected reference level */
659  switch (_selectedReferenceLevel)
660  {
661  case AdcReferenceLevel_1p5_VOLTS:

663  break;
664 
665  case AdcReferenceLevel_2p0_VOLTS:
666  refCalFactor = _refCalibrationInfo.ref20VFactor;
667  break;
668 
669  case AdcReferenceLevel_2p5_VOLTS:
670  refCalFactor = _refCalibrationInfo.ref25VFactor;
671  break;
672 
673  case AdcReferenceLevel_DONT_CARE:
674  ASSERT(false, ErrorLedMode_SOFTWARE);
675  break;
676 
677  default:
678  ASSERT(false, ErrorLedMode_SOFTWARE);
679  break;
680  }
681 
682  /* Loop through all values */
683  for (i = 0u; i < numValuesToCorrect; ++i)
684  {
685  /* Get the raw value */
686  value = rawValues[i];
687  ASSERT(value <= ADC_MAX_RESULT, ErrorLedMode_SOFTWARE);
688 
689  if ((correction == AdcCorrection_REF) ||
690  (correction == AdcCorrection_REF_GAIN_OFFSET))
691  {
692  /* Apply internal voltage reference factor */
693  value = value * 2u;
694  value = (uint16_t) ((((uint32_t) value) * ((uint32_t) refCalFactor)) >> 16);
695  }
696 
697  if ((correction == AdcCorrection_GAIN_OFFSET) ||
698  (correction == AdcCorrection_REF_GAIN_OFFSET))
699  {
700  /* Apply ADC gain factor */
701  value = value * 2u;
702  value = (uint16_t) ((((uint32_t) value) * ((uint32_t) _adcCalibrationInfo.gainFactor)) >> 16);
703 
704  /* Apply ADC offset */
705  resultSigned = (int16_t) value;
706  resultSigned = resultSigned + _adcCalibrationInfo.offset;
707  if (resultSigned > 0)
708  {
709  value = (uint16_t) resultSigned;
710  }
711  else
712  {
713  value = 0u;
714  }
715  }
716 
717  /* Store the corrected value */
718  correctedValues[i] = value;
719  }
