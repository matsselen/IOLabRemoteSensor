140 void Uart_Init(
141  UartPort_t port,
142  const UartPortConfig_t* config)
143 {
145  static volatile uint8_t* const _uartControlZeroRegisters[UART_NUM_PORTS] =
146  {
147  &UCA0CTL0,
148  &UCA1CTL0
149  };
151  static volatile uint16_t* const _uartBaudRateRegisters[UART_NUM_PORTS] =
152  {
153  &UCA0BRW,
154  &UCA1BRW
155  };
157  static volatile uint8_t* const _uartModulationRegisters[UART_NUM_PORTS] =
158  {
159  &UCA0MCTL,
160  &UCA1MCTL
161  };
163  static volatile uint8_t* const _uartAutoBaudRateRegisters[UART_NUM_PORTS] =
164  {
165  &UCA0ABCTL,
166  &UCA1ABCTL
167  };
169  static volatile uint8_t* const _uartIrdaTransmitRegisters[UART_NUM_PORTS] =
170  {
171  &UCA0IRTCTL,
172  &UCA1IRTCTL
173  };
175  static volatile uint8_t* const _uartIrdaReceiveRegisters[UART_NUM_PORTS] =
176  {
177  &UCA0IRRCTL,
178  &UCA1IRRCTL
179  };
180 
181  UartBaudRate_t selectedBaudRate;
182  UartParity_t selectedParity;
183  UartEndieness_t selectedEndieness;
184  UartStopBit_t selectedStopBit;
185 
186  /* Reset module */
187  *_uartControlOneRegisters[port] = UCSWRST;
188 
189  /* 8-bit data, UART mode, asynchronous mode*/
190  *_uartControlZeroRegisters[port] = 0u;
191 
192  selectedParity = config->parity;
193  switch (selectedParity)
194  {
195  case UartParity_EVEN:
196  /* Enable parity, Even parity, */
197  *_uartControlZeroRegisters[port] |= UCPEN | UCPAR;
198  break;
199 
200  case UartParity_ODD:
201  /* Enable parity, Even parity, 8-bit data, UART mode, asynchronous mode*/
202  *_uartControlZeroRegisters[port] |= UCPEN;
203  *_uartControlZeroRegisters[port] &= ~UCPAR;
204  break;
205 
206  case UartParity_NONE:
207  /* Enable parity, Even parity, 8-bit data, UART mode, asynchronous mode*/
208  *_uartControlZeroRegisters[port] &= ~UCPEN;
209  break;
210 
211  default:
212  ASSERT(false, ErrorLedMode_SOFTWARE);
213  break;
214  }
215 
216  selectedEndieness = config->endieness;
217  switch(selectedEndieness)
218  {
219  case UartEndieness_LsbFirst:
220  /* Set LSB first */
221  *_uartControlZeroRegisters[port] &= ~UCMSB;
222  break;
223 
224  case UartEndieness_MsbFirst:
225  /* set MSB first */
226  *_uartControlZeroRegisters[port] |= UCMSB;
227  break;
228 
229  default:
230  ASSERT(false, ErrorLedMode_SOFTWARE);
231  break;
232  }
233 
234  selectedStopBit = config->stopBits;
235  switch(selectedStopBit)
236  {
237  case UartStopBit_ONE:
238  *_uartControlZeroRegisters[port] &= ~UCSPB;
239  break;
240 
241  case UartStopBit_TWO:
242  *_uartControlZeroRegisters[port] |= UCSPB;
243  break;
244 
245  default:
246  ASSERT(false, ErrorLedMode_SOFTWARE);
247  break;
248  }
249 
250  /* Select ACLK as clock source,
251  disable Rx error and Rx break interrupts,
252  USCI module is not dormant,
253  next frame is normal data,
254  keep module in reset
255  */
256  *_uartControlOneRegisters[port] = UCSSEL__ACLK | UCSWRST;
257 
258  /* Disable auto baud rate detection. */
259  *_uartAutoBaudRateRegisters[port] = 0u;
260 
261  if (config->enableIrda)
262  {
263  /* Disable receive filter
264  Polarity set such that a low pulse indicates a light pulse was seen
265  */
266  *_uartIrdaReceiveRegisters[port] = UCIRRXPL;
267 
268  /* Set transmit pulse length to 3/16 of bit time
269  Use BITCLK16
270  Enable IrDA encoder/decoder
271  */
272  *_uartIrdaTransmitRegisters[port] = (5u * UCIRTXPL0) | UCIRTXCLK | UCIREN;
273  }
274  else
275  {
276  *_uartIrdaReceiveRegisters[port] = 0u;
277  *_uartIrdaTransmitRegisters[port] = 0u;
278  }
279 
280  /* Set baud rate (clock source = ACLK = 10 MHz).
281  In oversampling mode (UCOS16 = 1), baud rate regs (UCBRx) = INT(N/16)
282  and first stage modulator (UCBRFx) = round(((N/16) - INT(N/16)) * 16)
283  where N = (ACLK / baud rate).
284  */
285  selectedBaudRate = config->baudRate;
286  switch (selectedBaudRate)
287  {
288  case UartBaudRate_230400:
289  *_uartBaudRateRegisters[port] = 3u;
290  *_uartModulationRegisters[port] = (4u * UCBRF0) | UCOS16;
291  break;
292 
293  case UartBaudRate_115200:
294  *_uartBaudRateRegisters[port] = 6u;
295  *_uartModulationRegisters[port] = (8u * UCBRF0) | UCOS16;
296  break;
297 
298  case UartBaudRate_57600:
299  *_uartBaudRateRegisters[port] = 13u;
300  *_uartModulationRegisters[port] = (0u * UCBRF0) | UCOS16;
301  break;
302 
303  case UartBaudRate_38400:
304  *_uartBaudRateRegisters[port] = 19u;
305  *_uartModulationRegisters[port] = (9u * UCBRF0) | UCOS16;
306  break;
307 
308  case UartBaudRate_19200:
309  *_uartBaudRateRegisters[port] = 39u;
310  *_uartModulationRegisters[port] = (1u * UCBRF0) | UCOS16;
311  break;
312 
313  case UartBaudRate_9600:
314  *_uartBaudRateRegisters[port] = 78u;
315  *_uartModulationRegisters[port] = (2u * UCBRF0) | UCOS16;
316  break;
317 
318  default:
319  ASSERT(false, ErrorLedMode_SOFTWARE);
320  break;
321  }
322 }

329 void Uart_Open(UartPort_t port)
330 {
331  bool portIsOpen = _uartPortIsOpen[port];
332  volatile uint8_t* controlReg = _uartControlOneRegisters[port];
333 
334  ASSERT(portIsOpen == false, ErrorLedMode_SOFTWARE);
335 
336  /* Reset module */
337  *controlReg = *controlReg | UCSWRST;
338 
339  /* Configure pins to be used by peripheral */
340  Gpio_Configure(_uartGpioPorts[port], _uartGpioTransmitPinmasks[port],
341  GpioType_PERIPHERAL_OUTPUT, GpioValue_HIGH, GpioDriveStrength_DEFAULT);
342  Gpio_Configure(_uartGpioPorts[port], _uartGpioReceivePinmasks[port],
343  GpioType_PERIPHERAL_INPUT, GpioValue_DONT_CARE, GpioDriveStrength_DEFAULT);
344 
345  /* Enable a pull-up resister on the receive pin to prevent a floating input
346  from causing invalid bytes to be received.
347  */
348  Gpio_EnableResistor(_uartGpioPorts[port], _uartGpioReceivePinmasks[port], GpioResistorType_PULLUP);
349 
350  /* Flush transmit and receive buffers */
351  _uartBufferTxHead[port] = _uartBufferTxTail[port];
352  _uartBufferRxTail[port] = _uartBufferRxHead[port];
353 
354  /* Release module from reset */
355  *controlReg = *controlReg & ((uint8_t) ~(UCSWRST));
356 
357  _uartPortIsOpen[port] = true;
358 }
