542  Gpio_RegisterIsr(ACCELEROMETER_INTERRUPT_1_PORT, ACCELEROMETER_INTERRUPT_1_PIN,

556 bool Accelerometer_Config(const AccConfig_t* config)
557 {
558  bool success = false;
559  uint8_t systemStatus = 0u;
560  I2cError_t i2cResult = I2cError_OK;
561 
562  ASSERT(config != NULL, ErrorLedMode_SOFTWARE);
563  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
564 
565  /* store the initialization data */
566  _parameters.odr = config->odr;
567  _parameters.overSampleMode = config->overSampleMode;
568  _parameters.range = config->range;
569  _parameters.resolution = config->resolution;
570 
571  /* setup the configuration registers */
572  _ctrlRegOne = 0u;
573  _ctrlRegTwo = 0u;
574  _ctrlRegThree = 0u;
575  _ctrlRegFour = 0u;
576  _ctrlRegFive = 0u;
577  _xyzDataConfig = 0u;
578 
579  switch(config->range)
580  {
581  case AccFullScaleRange_8g:
582  _xyzDataConfig |= XyzDataConfig_8g;
583  break;
584 
585  case AccFullScaleRange_4g:
586  _xyzDataConfig |= XyzDataConfig_4g;
587  break;
588 
589  case AccFullScaleRange_2g:
590  _xyzDataConfig |= XyzDataConfig_2g;
591  break;
592 
593  default:
594  ASSERT(false, ErrorLedMode_SOFTWARE);
595  break;
596  }
597 
598  switch(config->odr)
599  {
600  case AccOdr_800:
601  _ctrlRegOne |= CtrlRegOne_ODR_800;
602  break;
603 
604  case AccOdr_400:
605  _ctrlRegOne |= CtrlRegOne_ODR_400;
606  break;
607 
608  case AccOdr_200:
609  _ctrlRegOne |= CtrlRegOne_ODR_200;
610  break;
611 
612  case AccOdr_100:
613  _ctrlRegOne |= CtrlRegOne_ODR_100;
614  break;
615 
616  case AccOdr_50:
617  _ctrlRegOne |= CtrlRegOne_ODR_50;
618  break;
619 
620  case AccOdr_12_5:
621  _ctrlRegOne |= CtrlRegOne_ODR_12_5;
622  break;
623 
624  case AccOdr_6_25:
625  _ctrlRegOne |= CtrlRegOne_ODR_6_25;
626  break;
627 
628  case AccOdr_1_56:
629  _ctrlRegOne |= CtrlRegOne_ODR_1_56;
630  break;
631 
632  default:
633  ASSERT(false, ErrorLedMode_SOFTWARE);
634  break;
635  }
636 
637  switch(config->resolution)
638  {
639  case AccResolution_14bit:
640  /* register should already be clear for this bit */
641  break;
642 
643  case AccResolution_8bit:
644  _ctrlRegOne |= CtrlRegOne_F_READ;
645  break;
646 
647  default:
648  ASSERT(false, ErrorLedMode_SOFTWARE);
649  }
650 
651  switch(config->overSampleMode)
652  {
653  case AccOverSample_Normal:
654  _ctrlRegTwo |= CtrlRegTwo_MODS_NRML;
655  break;
656 
657  case AccOverSample_HighRes:
658  _ctrlRegTwo |= CtrlRegTwo_MODS_HRES;
659  break;
660 
661  case AccOverSample_LNLP:
662  _ctrlRegTwo |= CtrlRegTwo_MODS_LNLP;
663  break;
664 
665  case AccOverSample_LowPower:
666  _ctrlRegTwo |= CtrlRegTwo_MODS_LPWR;
667  break;
668 
669  default:
670  ASSERT(false, ErrorLedMode_SOFTWARE);
671  }
672 
673  /* set the interrupt polarity as active high */
674  _ctrlRegThree |= CtrlRegThree_IPOL;
675 
676  /* enable the data ready interrupt */
677  _ctrlRegFour |= CtrlRegFour_INT_EN_DRDY;
678 
679  /* route the data ready interrupt to INT0 */
680  _ctrlRegFive |= CtrlRegFive_INT_CFG_DRDY;
681 
682  /* make sure the accelerometer is in idle mode or the settings won't take */
683  systemStatus = _AccReadSingleRegister(AccReg_SYSMOD);
684 
685  if ((systemStatus & SysMod_SYSMOD) == SysMod_STNDBY)
686  {
687  /* now send the configuration settings */
688  i2cResult = _AccWriteSingleRegister(AccReg_XYZ_DATA_CONFIG, _xyzDataConfig);
689  ASSERT(i2cResult == I2cError_OK, ErrorLedMode_SOFTWARE);
690 
691  i2cResult = _AccWriteSingleRegister(AccReg_CTRL_REG1, _ctrlRegOne);
692  ASSERT(i2cResult == I2cError_OK, ErrorLedMode_SOFTWARE);
693 
694  i2cResult = _AccWriteSingleRegister(AccReg_CTRL_REG2, _ctrlRegTwo);
695  ASSERT(i2cResult == I2cError_OK, ErrorLedMode_SOFTWARE);
696 
697  i2cResult = _AccWriteSingleRegister(AccReg_CTRL_REG3, _ctrlRegThree);
698  ASSERT(i2cResult == I2cError_OK, ErrorLedMode_SOFTWARE);
699 
700  i2cResult = _AccWriteSingleRegister(AccReg_CTRL_REG4, _ctrlRegFour);
701  ASSERT(i2cResult == I2cError_OK, ErrorLedMode_SOFTWARE);
702 
703  i2cResult = _AccWriteSingleRegister(AccReg_CTRL_REG5, _ctrlRegFive);
704  ASSERT(i2cResult == I2cError_OK, ErrorLedMode_SOFTWARE);
705 
706  success = true;
707  }
708 
709  _configured = true;
710 
711  return success;
712 }

714 /***************************************************************************//****************************************************************************/
717 void Accelerometer_Start(void)
718 {
719  if (_configured == false)
720  {
721  DEBUG("acc not conf");
722  }
723 
724  I2cError_t i2cError = I2cError_OK;
725  static uint8_t dummyData[BYTES_PER_SAMPLE] = {0};
726 
727  /* make sure the accelerometer is initialized */
728  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
729 
730  /* put the accelerometer in standby mode to make sure the settings take */
731  i2cError = _AccWriteSingleRegister(AccReg_CTRL_REG1, _ctrlRegOne);
732  ASSERT(i2cError == I2cError_OK, ErrorLedMode_SOFTWARE);
733 
734  /* read out the current data registers to clear the data ready */
735  _AccReadMultipleRegisters(AccReg_OUT_X_MSB, BYTES_PER_SAMPLE, dummyData);
736 
737  /* set the interrupt to trigger when data is ready */
738  _active = true;
739  GPIO_INT_ENABLE(ACCELEROMETER_INTERRUPT_0_PORT, ACCELEROMETER_INTERRUPT_0_PIN);
740 
741  /* set the accelerometer to active mode */
742  _ctrlRegOne |= CtrlRegOne_ACTIVE;
743 
744  /* write the active settings to the magnetometer */
745  i2cError = _AccWriteSingleRegister(AccReg_CTRL_REG1, _ctrlRegOne);
746  ASSERT(i2cError == I2cError_OK, ErrorLedMode_SOFTWARE);
747 }

820  bool sampleObtained = false;
821  uint8_t sampleArray[BYTES_PER_SAMPLE] = {0u};
822 
823  ASSERT(sample != NULL, ErrorLedMode_SOFTWARE);
824 
825  /* fast read mode is not supported, so only 14 bit samples are currently supported */
826  ASSERT(_parameters.resolution == AccResolution_14bit, ErrorLedMode_SOFTWARE);
827 
828  if (_dataReady)
829  {
830  /* clear the data ready flag before reading the data because the chip
831  interrupt will be cleared when the first byte is read. If the data
832  ready flag is cleared after the first byte is read, it could be a new
833  data ready that is being cleared.
834  */
835  _dataReady = false;
836 
837  _AccReadMultipleRegisters(AccReg_OUT_X_MSB, BYTES_PER_SAMPLE, sampleArray);
838  sample->xData = (int16_t)((((uint16_t)sampleArray[0u])<<8u) + sampleArray[1u]);
839  sample->yData = (int16_t)((((uint16_t)sampleArray[2u])<<8u) + sampleArray[3u]);
840  sample->zData = (int16_t)((((uint16_t)sampleArray[4u])<<8u) + sampleArray[5u]);
841  sampleObtained = true;
842  }
843 
844  return sampleObtained;
845 }
