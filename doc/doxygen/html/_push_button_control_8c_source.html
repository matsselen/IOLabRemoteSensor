156  if (GPIO_IS_LOW(BUTTON_1_PORT, BUTTON_1_PIN))

160  else
161  {
162  _buttonInfo[PushButton_ONE].buttonPressedDuration = 0u;
163  }
164 
165  /* If button 0 or 1 was pressed (debounced) and is now not pressed,
166  send event for the button release. */
167  if ((_buttonInfo[PushButton_ZERO].buttonPressedDuration >= BUTTON_DEBOUNCE_MS) &&
168  (_buttonInfo[PushButton_ZERO].previousButtonPressedDuration < BUTTON_DEBOUNCE_MS))
169  {
170  EventQueue_Add(Event_BUTTON_0_PRESSED, EventPriority_LOWEST);
171  }

173  (_buttonInfo[PushButton_ZERO].buttonPressedDuration == 0u))
174  {
175  EventQueue_Add(Event_BUTTON_0_RELEASED, EventPriority_LOWEST);
176  }
177
