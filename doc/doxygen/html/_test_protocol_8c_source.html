175 /***************************************************************************//****************************************************************************/
180 static void _HandleHumanPackets(SerialBuffer_t *packetReceived)
181 {
182  char_t* command;
183  uint16_t commandIndex;
184  int16_t compareResult;
185 
186  ASSERT(packetReceived != NULL, ErrorLedMode_SOFTWARE);
187 
188  command = strtok((char_t*) packetReceived->buffer, HUMAN_PARAM_SEPARATORS);
189 
190  if (command == NULL)
191  {
192  _SendNakHuman(NakErrorCode_NO_COMMAND);
193  }
194  else
195  {
196  /* Determine which command it was */
197  for (commandIndex = 0u; commandIndex < NUM_HUMAN_COMMANDS; ++commandIndex)
198  {
199  compareResult = strncmp(command, _humanCommands[commandIndex].commandString,
200  HUMAN_COMMAND_MAX_LENGTH);
201  if (compareResult == 0)
202  {
203  break;
204  }
205  }
206 
207  /* if the for-loop didn't break, then the command wasn't found */
208  if (commandIndex < NUM_HUMAN_COMMANDS)
209  {
210  /* call handler for matching command. It is up to the handler to
211  send the Ack.
212  */
213  DEBUG(command);
214  (*(_humanCommands[commandIndex].handlerFunction))();
215  }
216  else
217  {
218  _SendNakHuman(NakErrorCode_INVALID_COMMAND);
219  }
220  }
221 
222  /* Packet has been evaluated and can be released */
223  packetReceived->inUse = false;
224 }
