75 void Encoder_Init(void)
76 {
77  /* set the timer to use the external pin to clock it, have no divider, and count
78  continuously.
79  */
80  Timer_A2_Configure(TimerSource_EXTERNAL, TimerDivider_1, TimerRunMode_CONTINUOUS, 0u);
81  /* Set the timer to capture on both edges (rising and falling), use the CCxA input source,
82  DO NOT synchronize the capture with the timer, use capture mode (not compare). Synchronizing
83  will make the capture wait until the next edge of the timer which might never occur depending
84  on the movement of the encoder, so non-synchronous is desired.
85 
86  CC0 is for the direction change and CC1 is for the software capture. Software capture uses
87  VCC and GND as capture inputs and switches between them to generate a capture event.
88  */
89  Timer_A2CCX_Configure(TimerA2CC_0, (CM_3 | CCIS__CCIxA | CAP), 0);
90  Timer_A2CCX_Configure(TimerA2CC_1, (CM_3 | CCIS1 | CAP), 0);
91  _initialized = true;
92 }

138 EncoderStatus_t Encoder_Read(
139  int16_t counts[],
140  uint16_t *elements,
141  uint16_t maxElements)
142 {
143  EncoderStatus_t status;
144  int16_t index;
145  bool encoderInteruptEnabled = TIMER_A2_CC0_INT_IS_ENABLED();
146 
147  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
148  ASSERT(counts != NULL, ErrorLedMode_SOFTWARE);
149  ASSERT(elements != NULL, ErrorLedMode_SOFTWARE);
150 
151  /* trigger a capture to get the most recent count */
152  TIMER_A2_CC1_TRIGGER;
153 
154  /* disable the encoder interrupts so the data can't be changed
155  while it's being copied.
156  */
157  if (encoderInteruptEnabled)
158  {
159  TIMER_A2_CC0_INT_DISABLE();
160  }
161 
162  if (maxElements >= NUM_DIRECTION_CHANGE)
163  {
164  /* copy the encoder data elements then reset the
165  storage buffer.
166  */
167  for (index = NUM_DIRECTION_CHANGE - 1; index >= 0; --index)
168  {
169  counts[index] = _encoderCounts[index];
170  }
171 
172  if (_encoderCountIndex == 0)
173  {
174  *elements = 1u;
175  }
176  else
177  {
178  *elements = _encoderCountIndex;
179  }
180 
181  status = _currentStatus;
182 
183  _Encoder_ClearBuffer();
184  }
185  else
186  {
187  status = EncoderStatus_OTHER_ERROR;
188  }
189 
190  if (encoderInteruptEnabled)
191  {
192  TIMER_A2_CC0_INT_ENABLE();
193  }
194 
195  _currentStatus = EncoderStatus_OKAY;
196  return status;
197 }

206 static void _Encoder_AddSample(uint16_t counts, EncoderDirection_t direction)
207 {
208  int16_t *sampleLocation = NULL;
209 
210  /* if the direction is the same, we can just add/subtract to the value
211  in that buffer location. If the direction is different, we have to
212  move to the next location in the buffer if there is room.
213  */
214  if ((direction == _previousDirection) ||
215  ((_encoderCountIndex == 0u) && (_previousDirection == EncoderDirection_INVALID)))
216  {
217  sampleLocation = &_encoderCounts[_encoderCountIndex];
218  }
219  else
220  {
221  ++_encoderCountIndex;
222  if (_encoderCountIndex < NUM_DIRECTION_CHANGE)
223  {
224  sampleLocation = &_encoderCounts[_encoderCountIndex];
225  }
226  else
227  {
228  _currentStatus = EncoderStatus_BUFFER_OVERFLOW;
229  }
230  }
231 
232  /* make sure there was room in the buffer before writing to it */
233  if (sampleLocation != NULL)
234  {
235  /* add the new counts to the buffer with the correct sign */
236  if (direction == EncoderDirection_FORWARD)
237  {
238  *sampleLocation += (int16_t)(counts - _previousCount);
239  }
240  else if (direction == EncoderDirection_BACKWARD)
241  {
242  *sampleLocation -= (int16_t)(counts - _previousCount);
243  }
244  }
245 
246  /* update the current values to previous values */
247  _previousCount = counts;
248  _previousDirection = direction;
249 }

312 __interrupt void _TimerA2CCxIsr(void)
313 {
314  static volatile uint16_t currentCount;
315  static EncoderDirection_t currentDirection;
316 
317  /* Note: intrinsic allows compiler to optimize switch */
318  switch (_even_in_range(TA2IV, TA2IV_TA2IFG))
319  {
320  case TA2IV_NONE:
321  /* Do nothing. No interrupt to handle.
322  This case may occur if an interrupt is disabled at the same time
323  as the interrupt flag is set. The CPU is already committed to
324  running the ISR, but the interrupt has been disabled, which
325  keeps the interrupt vector register from indicating the flag
326  has been set.
327  */
328  break;
329 
330  /* CC1 interrupt flag is set */
331  case TA2IV_TA2CCR1:
332  if (TIMER_A2_CC1_OVERFLOWED == false)
333  {
334  currentCount = TIMER_A2_CC1_GET;
335  /* the CC0 pin has the direction signal */
336  if (TIMER_A2_CC0_PIN_IS_HIGH)
337  {
338  currentDirection = EncoderDirection_BACKWARD;
339  }
340  else
341  {
342  currentDirection = EncoderDirection_FORWARD;
343  }
344 
345  _Encoder_AddSample(currentCount, currentDirection);
346  }
347  else
348  {
349  TIMER_A2_CC1_CLEAR_OVERFLOW;
350  _currentStatus = EncoderStatus_SW_CAPTURE_OVERFLOW;
351  }
352  break;
353 
354  /* CC2 interrupt flag is set */
355  case TA2IV_TA2CCR2:
356  /* do nothing */
357  break;
358 
359  /* TA interrupt flag is set */
360  case TA2IV_TA2IFG:
361  /* do nothing */
362  break;
363 
364  default:
365  _never_executed();
366  break;
367  }
368 
369  _low_power_mode_off_on_exit();
