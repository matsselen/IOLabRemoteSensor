60 void Clock_Init(void)
61 {
62  uint16_t timeout;
63 
64  /* Turn off FLL. Locks DCO at current frequency. */
65  _bis_SR_register(SCG0);
66 
67  /* Crystal oscillator settings:
68  Turn off XT1.
69  Drive strength level 2 for XT2 (operating range of 16 to 24 MHz).
70  High frequency mode for XT2.
71  No bypass for XT2.
72  SMCLK on.
73  XT2 on.
74  */
75  UCSCTL6 = XT2DRIVE_2 | XT1OFF;
76 
77  /* Enable XT2 oscillator pins. */
78  Gpio_Configure(OSCILLATOR_IN_PORT, OSCILLATOR_IN_PIN,
79  GpioType_PERIPHERAL_INPUT, GpioValue_DONT_CARE, GpioDriveStrength_DEFAULT);
80  Gpio_Configure(OSCILLATOR_OUT_PORT, OSCILLATOR_OUT_PIN,
81  GpioType_PERIPHERAL_OUTPUT, GpioValue_LOW, GpioDriveStrength_DEFAULT);
82 
83  /* Wait for oscillator input to stabilize or a timeout expires. */
84  timeout = OSC_STABILIZE_TIMEOUT_USECS / OSC_STABILIZE_LOOP_INTERVAL_USECS;
85  while (((UCSCTL7 & XT2OFFG) > 0u) && (timeout > 0u))
86  {
87  /* Clear oscillator fault flag and wait. */
88  UCSCTL7 &= ~(XT2OFFG);
89 
90  _delay_cycles(OSC_STABILIZE_LOOP_INTERVAL_USECS);
91  --timeout;
92  }
93 
94  ASSERT(timeout > 0u, ErrorLedMode_SOFTWARE);
95 
96  /* Set DCO to lowest range and disable modulation so save power since it isn't used */
97  UCSCTL1 = DCORSEL_0;
98 
99  /* We are not using the FLL so leave it default and off */
100  // no code for this
101 
102  /* Clock source settings:
103  Ext ACLK = XT2CLK / 32
104  ACLK = XT2CLK / 2
105  SMCLK = XT2CLK / 1
106  MCLK = XT2CLK / 1
107  */
108  UCSCTL4 = SELA__XT2CLK | SELS__XT2CLK | SELM__XT2CLK;
109  UCSCTL5 = DIVPA__32 | DIVA__2 | DIVS__1 | DIVM__1;
110 
111  /* Enable conditional requests for SMCLK, MCLK, and ACLK. */
112  UCSCTL8 = SMCLKREQEN | MCLKREQEN | ACLKREQEN;
113 
114  /* Clear DCO and all oscillator fault flags. */
115  UCSCTL7 = 0u;
116 
117  /* Clear, then enable oscillator fault interrupt. */
118  SFRIFG1 &= ~(OFIFG);
119  SFRIE1 |= OFIE;
120 }
