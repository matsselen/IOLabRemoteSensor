59 void Pmm_Init(void)
60 {
61  PmmStatus_t status;
62 
63  /* set the system to core level 3 to allow full speed operation of 24 MHz with Vdd > 2.7V */
64  status = _SetVCore(3u);
65  ASSERT(status == PmmStatus_OKAY, ErrorLedMode_SOFTWARE);
66 
67  /* Open PMM registers for write access */
68  PMMCTL0_H = PMM_PASSWORD;
69 
70  /* Disable interrupts and clear flags */
71  PMMRIE = 0u;
72  PMMIFG = 0u;
73 
74  /* Turn off low-side SVS and SVM.
75  Mask events just in case.
76  */
77  SVSMLCTL &= ~(SVMLFP | SVMLE | SVMLOVPE |
78  SVSLFP | SVSLE | SVSMLACE | SVSLMD);
79  SVSMLCTL |= SVSMLEVM;
80 
81  /* Turn off high-side SVM.
82  Turn on high-side SVS.
83  */
84  SVSMHCTL &= ~(SVMHFP | SVMHE | SVMHOVPE);
85  SVSMHCTL |= SVSHE;
86 
87  /* Clear high-side delay flag */
88  PMMIFG &= ~(SVSMHDLYIFG);
89 
90  /* Put high-side SVS in full performance mode with automatic hardware control
91  and active in LPM 2/3/4 (causes SVS to enter normal mode in LPM 2/3/4).
92  Ensure events are not masked.
93  */
94  SVSMHCTL |= (SVSHFP | SVSMHACE | SVSHMD);
95  SVSMHCTL &= ~(SVSMHEVM);
96 
97  /* Wait until high-side SVM (and SVS) is settled */
98  while ((PMMIFG & SVSMHDLYIFG) == 0u)
99  {
100  /* Do nothing */
101  }
102 
103  /* Clear flags */
104  PMMIFG = 0u;
105 
106  /* Enable power-on reset (POR) if high-side voltage drops too low */
107  PMMRIE = SVSHPE;
108 
109  /* Lock PMM registers to prevent write access */
110  PMMCTL0_H = 0x00u;
111 }

120 static PmmStatus_t _SetVCore(uint8_t level)
121 {
122  uint8_t actualLevel;
123  PmmStatus_t status = PmmStatus_OKAY;
124 
125  ASSERT(level <= PMM_MAX_VCORE_LEVEL, ErrorLedMode_SOFTWARE);
126 
127  /* Get current Vcore level */
128  actualLevel = (uint8_t) (PMMCTL0 & PMMCOREV_3);
129 
130  /* Increase/decrease level one step at a time */
131  while ((level != actualLevel) && (status == PmmStatus_OKAY))
132  {
133  if (level > actualLevel)
134  {
135  ++actualLevel;
136  status = _SetVCoreUp(actualLevel);
137  }
138  else
139  {
140  --actualLevel;
141  status = _SetVCoreDown(actualLevel);
142  }
143  }
144 
145  return status;
146 }

155 static PmmStatus_t _SetVCoreUp(uint8_t level)
156 {
157  uint16_t PMMRIE_backup;
158  uint16_t SVSMHCTL_backup;
159  uint16_t SVSMLCTL_backup;
160  PmmStatus_t status = PmmStatus_OKAY;
161 
162  ASSERT(level <= PMM_MAX_VCORE_LEVEL, ErrorLedMode_SOFTWARE);
163 
164  /* The code flow for increasing the Vcore has been altered to work around
165  the erratum FLASH37.
166  Please refer to the Errata sheet to know if a specific device is affected
167  DO NOT ALTER THIS FUNCTION
168  */
169 
170  /* Open PMM registers for write access */
171  PMMCTL0_H = PMM_PASSWORD;
172 
173  /* Disable dedicated interrupts */
174  /* Backup all registers */
175  PMMRIE_backup = PMMRIE;
176  PMMRIE &= ~(SVMHVLRPE | SVSHPE | SVMLVLRPE | SVSLPE | SVMHVLRIE |
177  SVMHIE | SVSMHDLYIE | SVMLVLRIE | SVMLIE | SVSMLDLYIE);
178  SVSMHCTL_backup = SVSMHCTL;
179  SVSMLCTL_backup = SVSMLCTL;
180 
181  /* Clear flags */
182  PMMIFG = 0u;
183 
184  /* Set SVM highside to new level and check if a VCore increase is possible */
185  SVSMHCTL = SVMHE | SVSHE | (SVSMHRRL0 * level);
186 
187  /* Wait until SVM highside is settled */
188  while ((PMMIFG & SVSMHDLYIFG) == 0u)
189  {
190  /* Do nothing */
191  }
192 
193  /* Clear flag */
194  PMMIFG &= ~SVSMHDLYIFG;
195 
196  /* Check if a VCore increase is possible */
197  if ((PMMIFG & SVMHIFG) == SVMHIFG)
198  {
199  /* Vcc is too low for a Vcore increase */
200 
201  /* recover the previous settings */
202  PMMIFG &= ~SVSMHDLYIFG;
203  SVSMHCTL = SVSMHCTL_backup;
204 
205  /* Wait until SVM highside is settled */
206  while ((PMMIFG & SVSMHDLYIFG) == 0u)
207  {
208  /* Do nothing */
209  }
210 
211  status = PmmStatus_ERROR;
212  }
213  else
214  {
215  /* Vcc is high enough for a Vcore increase */
216 
217  /* Set also SVS highside to new level */
218  SVSMHCTL |= (SVSHRVL0 * level);
219 
220  /* Wait until SVM highside is settled */
221  while ((PMMIFG & SVSMHDLYIFG) == 0u)
222  {
223  /* Do nothing */
224  }
225 
226  /* Clear flag */
227  PMMIFG &= ~SVSMHDLYIFG;
228 
229  /* Set VCore to new level */
230  PMMCTL0_L = (uint8_t) (PMMCOREV0 * level);
231 
232  /* Set SVM, SVS low side to new level */
233  SVSMLCTL = SVMLE | (SVSMLRRL0 * level) | SVSLE | (SVSLRVL0 * level);
234 
235  /* Wait until SVM, SVS low side is settled */
236  while ((PMMIFG & SVSMLDLYIFG) == 0u)
237  {
238  /* Do nothing */
239  }
240 
241  /* Clear flag */
242  PMMIFG &= ~SVSMLDLYIFG;
243 
244  /* SVS, SVM low and high side are now set to protect for the new core level */
245 
246  /* Restore Low side settings
247  Clear all other bits _except_ level settings
248  */
249  SVSMLCTL &= (SVSLRVL0 + SVSLRVL1 + SVSMLRRL0 + SVSMLRRL1 + SVSMLRRL2);
250 
251  /* Clear level settings in the backup register, keep all other bits */
252  SVSMLCTL_backup &= ~(SVSLRVL0 + SVSLRVL1 + SVSMLRRL0 + SVSMLRRL1 + SVSMLRRL2);
253 
254  /* Restore low-side SVS monitor settings */
255  SVSMLCTL |= SVSMLCTL_backup;
256 
257  /* Restore High side settings
258  Clear all other bits except level settings
259  */
260  SVSMHCTL &= (SVSHRVL0 + SVSHRVL1 + SVSMHRRL0 + SVSMHRRL1 + SVSMHRRL2);
261 
262  /* Clear level settings in the backup register, keep all other bits */
263  SVSMHCTL_backup &= ~(SVSHRVL0 + SVSHRVL1 + SVSMHRRL0 + SVSMHRRL1 + SVSMHRRL2);
264 
265  /* Restore backup */
266  SVSMHCTL |= SVSMHCTL_backup;
267 
268  /* Wait until high side and low side are settled */
269  while ((PMMIFG & SVSMLDLYIFG) == 0u)
270  {
271  /* Do nothing */
272  }
273  while ((PMMIFG & SVSMHDLYIFG) == 0u)
274  {
275  /* Do nothing */
276  }
277  }
278 
279  /* Clear all flags */
280  PMMIFG &= ~(SVMHVLRIFG | SVMHIFG | SVSMHDLYIFG | SVMLVLRIFG | SVMLIFG | SVSMLDLYIFG);
281 
282  /* Restore PMM interrupt enable register */
283  PMMRIE = PMMRIE_backup;
284 
285  /* Lock PMM registers to prevent write access */
286  PMMCTL0_H = 0x00u;
287 
288  return status;
289 }
