369 void _HandleStatusPacket(
370  uint8_t messageData[],
371  uint8_t messageLength)
372 {
373 #define STATUS_PACKET_EXPECTED_LENGTH ( 3u )
374 #define FIRMWARE_MAJOR_INDEX ( 0u )
375 #define FIRMWARE_MINOR_INDEX ( 1u )
376 #define CONNECTION_STATUS_INDEX ( 2u )
377 
378  ASSERT((messageData != NULL) && (messageLength == STATUS_PACKET_EXPECTED_LENGTH),
379  ErrorLedMode_SOFTWARE);
380 
381  _SetTxCommand(CommCommand_STATUS);
382 
383  /* Store the RF Status information */
384  _rfStatus.fwMajor = messageData[FIRMWARE_MAJOR_INDEX];
385  _rfStatus.fwMinor = messageData[FIRMWARE_MINOR_INDEX];
386  _rfStatus.connectionStatus = (RfConnectionStatus_t)messageData[CONNECTION_STATUS_INDEX];
387 
388  switch(_rfStatus.connectionStatus)
389  {
390  case RfConnectionStatus_DATA:
391  LedControl_SetLed(Led_1_GREEN, LedState_BLINK_FAST_ODD);
392  break;
393 
394  case RfConnectionStatus_CONNECTED:
395  LedControl_SetLed(Led_1_GREEN, LedState_BLINK_SLOW_ODD);
396  break;
397 
398  case RfConnectionStatus_PAIRED_NOT_CONNECTED:
399  LedControl_SetLed(Led_1_GREEN, LedState_BLINK_1);
400  break;
401 
402  case RfConnectionStatus_PAIRING:
403  LedControl_SetLed(Led_1_GREEN, LedState_BLINK_3);
404  break;
405 
406  case RfConnectionStatus_NOT_PAIRED:
407  LedControl_SetLed(Led_1_GREEN, LedState_ON);
408  break;
409 
410  default:
411  LedControl_SetLed(Led_1_GREEN, LedState_OFF);
412  LedControl_SetLed(Led_1_RED, LedState_BLINK_FAST_ODD);
413  break;
414  }
