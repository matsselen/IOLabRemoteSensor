147 bool Barometer_ReadSample(BarometerSample_t* sample)
148 {
149  volatile bool sampleSuccess = false;
150  uint8_t rawRegisters[BYTES_PER_SAMPLE] = {0u};
151 
152  ASSERT(sample != NULL, ErrorLedMode_SOFTWARE);
153 
154  if (_conversionInProgress)
155  {
156  /* clear the conversion flag for this sample */
157  _conversionInProgress = false;
158 
159  /* read the sample data from the barometer */
160  _BarReadMultipleRegisters(BarometerReg_PADC_MSB, BYTES_PER_SAMPLE, rawRegisters);
161 
162  /* stick the raw samples in the barometer sample */
163  sample->pressure = (uint16_t)((((uint16_t)rawRegisters[0u])<<8u) + rawRegisters[1u]);
164  sample->temperature = (uint16_t)((((uint16_t)rawRegisters[2u])<<8u) + rawRegisters[3u]);
165 
166  sampleSuccess = true;
167  }
168 
169  return sampleSuccess;
170 }

288 void _BarReadMultipleRegisters(BarometerReg_t regStartAddress,
289  uint8_t numberOfRegisters,
290  uint8_t values[])
291 {
292  volatile uint8_t regNum;
293 
294  ASSERT((values != NULL) && (numberOfRegisters >0), ErrorLedMode_SOFTWARE);
295 
296  BAROMETER_START_SPI_XFER;
297  CLOCK_DELAY_US(SPI_CS_DELAY_US);
298  for(regNum = 0; regNum < numberOfRegisters; regNum++)
299  {
300  Spi_ReadRegister((regStartAddress + (regNum<<1)), &values[regNum]);
301  }
302  /* the barometer needs an extra byte to reset the state machine, see
303  the device datasheet */
304  Spi_WriteByte(BLANK_BYTE);
305  CLOCK_DELAY_US(SPI_CS_DELAY_US);
306  BAROMETER_END_SPI_XFER;
307 }
