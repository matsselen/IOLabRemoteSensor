341 void Gyroscope_Init(void)
342 {
343  /* See if the gyroscope has already been initialized */
344  ASSERT(_initialized == false, ErrorLedMode_SOFTWARE);
345 
346  /* check to make sure we have the expected gyroscope using the
347  WHO_AM_I id.
348  */
349  volatile uint8_t receivedByte = 0u;
350  receivedByte = _GyroReadSingleRegister(GyroscopeReg_WHO_AM_I);
351  ASSERT(receivedByte == WHO_AM_I, ErrorLedMode_SOFTWARE);
352 
353  /* register the interrupt for the data ready pin */
354  Gpio_RegisterIsr(GYRO_INTERRUPT_1_PORT, GYRO_INTERRUPT_1_PIN,
355  &_HandleDataReadyInterrupt, GpioEdge_RISING);
356  Gpio_RegisterIsr(GYRO_INTERRUPT_0_PORT, GYRO_INTERRUPT_0_PIN,
357  &_HandleOtherInterrupt, GpioEdge_RISING);
358  _initialized = true;
359 
360 }

368 void Gyroscope_Config(const GyroConfig_t* config)
369 {
370 
371  ASSERT(config != NULL, ErrorLedMode_SOFTWARE);
372  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
373 
374  /* setup the configuration registers */
375  _ctrlRegOne = 0u;
376  _ctrlRegFour = 0u;
377 
378  switch(config->odr)
379  {
380  case GyroOdr_760:
381  _ctrlRegOne |= CtrlRegOne_ODR_760;
382  break;
383 
384  case GyroOdr_380:
385  _ctrlRegOne |= CtrlRegOne_ODR_380;
386  break;
387 
388  case GyroOdr_190:
389  _ctrlRegOne |= CtrlRegOne_ODR_190;
390  break;
391 
392  case GyroOdr_95:
393  _ctrlRegOne |= CtrlRegOne_ODR_95;
394  break;
395 
396  default:
397  ASSERT(false, ErrorLedMode_SOFTWARE);
398  break;
399  }
400 
401  switch(config->resolution)
402  {
403  case GyroResolution_2000:
404  _ctrlRegFour |= CtrlRegFour_RES_2000;
405  break;
406 
407  case GyroResolution_500:
408  _ctrlRegFour |= CtrlRegFour_RES_500;
409  break;
410 
411  case GyroResolution_250:
412  _ctrlRegFour |= CtrlRegFour_RES_250;
413  break;
414 
415  default:
416  ASSERT(false, ErrorLedMode_SOFTWARE);
417  break;
418  }
419 
420  /* enable the data ready interrupt */
421  _ctrlRegThree |= CtrlRegThree_I2_DRDY;
422 
423 
424  /* block data update when MSB and LSB read, Data MSB @ lower address */
425  _ctrlRegFour |= (CtrlRegFour_BDU | CtrlRegFour_BLE);
426 
427 
428  /* send the configuration settings */
429  _GyroWriteRegister(GyroscopeReg_CTRL_REG1, _ctrlRegOne);
430 
431  _GyroWriteRegister(GyroscopeReg_CTRL_REG3, _ctrlRegThree);
432

437 /***************************************************************************//****************************************************************************/
441 void Gyroscope_Start(void)
442 {
443 
444  static uint8_t dummyData[BYTES_PER_SAMPLE];
445 
446  /* make sure the accelerometer is initialized */
447  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
448 
449  /* make sure the gyro is disabled first */
450  _GyroWriteRegister(GyroscopeReg_CTRL_REG1, 0x07);
451 
452  /* read the data to clear the data ready pin */
453  _GyroReadMultipleRegisters(GyroscopeReg_OUT_X_L, BYTES_PER_SAMPLE, dummyData);
454 
455  /* set the interrupt to trigger when data is ready */
456  GPIO_INT_ENABLE(GYRO_INTERRUPT_1_PORT, GYRO_INTERRUPT_1_PIN);
457 
458  /* set the gyroscope to active mode */
459  _ctrlRegOne |= ( CtrlRegOne_PD | CtrlRegOne_Zen | CtrlRegOne_Yen | CtrlRegOne_Xen);
460  _GyroWriteRegister(GyroscopeReg_CTRL_REG1, _ctrlRegOne);
461 
462 }
463 
464 /***************************************************************************//****************************************************************************/
468 void Gyroscope_Stop(void)
469 {
470  /* clear the data ready interrupt trigger */
471  GPIO_INT_DISABLE(GYRO_INTERRUPT_1_PORT, GYRO_INTERRUPT_1_PIN);
472 
473  /* put the gyroscope into power down mode */
474  _ctrlRegOne &= ~CtrlRegOne_PD;
475 
476  /* write the settings to the gyroscope */
477  _GyroWriteRegister(GyroscopeReg_CTRL_REG1, _ctrlRegOne);
478

481 /***************************************************************************//****************************************************************************/
487 bool Gyroscope_ReadSample(GyroSample_t* sample)
488 {
489  bool sampleObtained = false;
490  uint8_t rawRegisters[BYTES_PER_SAMPLE] = {0u};
491 
492  ASSERT(sample != NULL, ErrorLedMode_SOFTWARE);
493 
494  /* read the sample data from the gyroscope */
495  _GyroReadMultipleRegisters(GyroscopeReg_OUT_X_L, BYTES_PER_SAMPLE, rawRegisters);
496 
497  /* stick the raw samples in the gyroscope sample */
498  sample->xData = (uint16_t)((((uint16_t)rawRegisters[0u])<<8u) + rawRegisters[1u]);
499  sample->yData = (uint16_t)((((uint16_t)rawRegisters[2u])<<8u) + rawRegisters[3u]);
500  sample->zData = (uint16_t)((((uint16_t)rawRegisters[4u])<<8u) + rawRegisters[5u]);
501  sampleObtained = true;
502  return sampleObtained;
503 }

505 /***************************************************************************//****************************************************************************/
513 void Gyroscope_ReadSampleArray(
514  uint16_t maxLength,
515  uint16_t* actualLength,
516  uint8_t data[])
517 {
518  ASSERT(data != NULL, ErrorLedMode_SOFTWARE);
519  ASSERT(maxLength >= BYTES_PER_SAMPLE, ErrorLedMode_SOFTWARE);
520 
521  if (_dataReady)
522  {
523  /* clear the data ready flag before reading the data because the chip
524  interrupt will be cleared when the first byte is read. If the data
525  ready flag is cleared after the first byte is read, it could be a new
526  data ready that is being cleared.
527  */
528  _dataReady = false;
529 
530  _GyroReadMultipleRegisters(GyroscopeReg_OUT_X_L, BYTES_PER_SAMPLE, data);
531  *actualLength = BYTES_PER_SAMPLE;
532  }
533  else
534  {
535  *actualLength = 0u;
536  }
537 }
