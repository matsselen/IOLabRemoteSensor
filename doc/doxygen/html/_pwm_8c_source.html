110 void Pwm_SetFrequency(uint32_t freq)
111 {
112  uint8_t channelNumber = 0;
113 
114  /* convert frequency to compare value */
115  _maxCount = (uint16_t)((((uint32_t)(PWM_CLK*10u) + (uint32_t)(freq/2))/freq));
116 
117  /* load compare value into the period register */
118  Timer_B0_Configure(TimerSource_SMCLK, TimerDivider_20, TimerCounterLength_16BITS, TimerRunMode_UP, _maxCount);
119 
120  /* update the duty cycles of any enabled channels */
121  for(channelNumber = 0; channelNumber < NUM_CHANNELS; channelNumber++)
122  {
123  if( _openChannels[channelNumber])
124  {
125  Pwm_SetDuty((PwmChannel_t)channelNumber, _dutyCycles[channelNumber]);
126  }
127  }
128 
129  if(_AnyChannelsEnabled())
130  {
131  TIMER_B0_START(TimerRunMode_UP);
132  }
133 }
