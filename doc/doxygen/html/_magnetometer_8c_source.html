320  break;
321  }
322 
323  switch(config->resolution)
324  {
325  case MagResolution_16BIT:
326  /* register should already be 0 for this bit */
327  break;
328 
329  case MagResolution_8BIT:

336  break;

338 
339  /* make sure the magnetometer is in idle mode, or the settings won't take */

353 
354  /* the config is only successful if all the registers were successful. */
355  if (i2cResult == I2cError_OK)
356  {
357  success = true;
358  }
359  }
360  }
361 
362  return success;
363 }

368 void Magnetometer_Start(void)
369 {
370  I2cError_t i2cError = I2cError_OK;
371 
372  /* make sure the magnetometer is initialized */
373  ASSERT(_initialized, ErrorLedMode_SOFTWARE);
374  ASSERT(_parameters.samplingMode == MagSampling_Auto, ErrorLedMode_SOFTWARE);
375 
376  /* make sure the accelerometer is in standby or the register settings

378  */
379  i2cError = _WriteSingleRegister(MagReg_CTRL_REG1, _controlRegOne);
380  ASSERT(i2cError == I2cError_OK, ErrorLedMode_SOFTWARE);
381 
382  /* read X axis MSB to clear the data ready interrupt */
383  (void) _ReadSingleRegister(MagReg_OUT_X_MSB);
384 
385  /* set the interrupt to trigger when data is ready */
386  GPIO_INT_ENABLE(MAGNETOMETER_INTERRUPT_PORT, MAGNETOMETER_INTERRUPT_PIN);
387 
388  /* set the magnetometer to active */
389  _controlRegOne |= ACTIVE_MODE_ENABLE;
390 
391  /* write the control settings back to the magnetometer */
392  i2cError = _WriteSingleRegister(MagReg_CTRL_REG1, _controlRegOne);
393  ASSERT(i2cError == I2cError_OK, ErrorLedMode_SOFTWARE);
394 }

419 bool Magnetometer_TriggerSampleBlocking(MagSample_t* sample)
420 {
421  #define BYTES_IN_SAMPLE 6u
422 
423  bool sampleSuccess = false;
424  volatile uint8_t controlRegOne = 0;
425  volatile uint8_t dataReady = 0;
426  uint8_t sampleArray[BYTES_IN_SAMPLE];
427  volatile uint8_t index;
428  for(index = 0; index < BYTES_IN_SAMPLE; index++)
429  {
430  sampleArray[index] = 0;
431  }
432 
433  ASSERT(sample != NULL, ErrorLedMode_SOFTWARE);
434 
435  controlRegOne = _ReadSingleRegister(MagReg_CTRL_REG1);
436  controlRegOne |= TRIGGER_MEASUREMENT;
437  _WriteSingleRegister(MagReg_CTRL_REG1, controlRegOne);
438 
439  volatile uint16_t limitCounter = UINT16_T_MAX;
440  while ((dataReady & DataReadyStatus_ZYXDR) == false)
441  {
442  dataReady = _ReadSingleRegister(MagReg_DR_STATUS);
443  limitCounter--;
444  ASSERT(limitCounter > 0, ErrorLedMode_SOFTWARE);
445  }
446 
447  _ReadMultipleRegisters(MagReg_OUT_X_MSB, BYTES_IN_SAMPLE, sampleArray);
448 
449  sample->xData = (int16_t)((((uint16_t)sampleArray[0u])<<8u) + sampleArray[1u]);
450  sample->yData = (int16_t)((((uint16_t)sampleArray[2u])<<8u) + sampleArray[3u]);
451  sample->zData = (int16_t)((((uint16_t)sampleArray[4u])<<8u) + sampleArray[5u]);
452 
453  sampleSuccess = true;
454  return sampleSuccess;
455 }

460 void Magnetometer_ResetSensorBlocking(void)
461 {
462  I2cError_t i2cStatus = I2cError_OK;
463  uint8_t volatile controlRegTwoCopy = _controlRegTwo;
464  uint16_t volatile timeoutCounter;
465 
466  /* set the sensor reset bit in control register two */
467  controlRegTwoCopy |= Mag_RST;

470  ASSERT(i2cStatus == I2cError_OK, ErrorLedMode_SOFTWARE);
471 
472  timeoutCounter = 1000;
473  while ((controlRegTwoCopy & Mag_RST) && (timeoutCounter > 0))
474  {

476  timeoutCounter--;
477  }
